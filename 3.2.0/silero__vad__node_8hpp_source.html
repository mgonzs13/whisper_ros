130 template <typename T>
131 std::vector<float> convert_to_float(const std::vector<T> &input) {
132 static_assert(std::is_integral<T>::value,
133 "Input type must be an integral type.");
134
135 std::vector<float> output;
136 output.reserve(input.size());
137
138 if constexpr (std::is_same<T, uint8_t>::value) {
139 // uint8_t data normalized to [0.0, 1.0]
140 for (T value : input) {
141 output.push_back(static_cast<float>(value) / 255.0f);
142 }
143 } else if constexpr (std::is_same<T, int8_t>::value) {
144 // int8_t data normalized to [-1.0, 1.0]
145 constexpr float scale = 1.0f / 127.0f; // Max positive value of int8_t
146 for (T value : input) {
147 output.push_back(static_cast<float>(value) * scale);
148 }
149 } else if constexpr (std::is_same<T, int16_t>::value) {
150 // int16_t data normalized to [-1.0, 1.0]
151 constexpr float scale = 1.0f / 32767.0f; // Max positive value of int16_t
152 for (T value : input) {
153 output.push_back(static_cast<float>(value) * scale);
154 }
155 } else if constexpr (std::is_same<T, int32_t>::value) {
156 // int32_t data normalized to [-1.0, 1.0]
157 constexpr float scale =
158 1.0f / 2147483647.0f; // Max positive value of int32_t
159 for (T value : input) {
160 output.push_back(static_cast<float>(value) * scale);
161 }
162 } else {
163 throw std::invalid_argument(
164 "Unsupported data type for audio conversion.");
165 }
166
167 return output;
168 }
